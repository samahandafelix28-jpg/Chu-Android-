/* ---------------- storage helpers ---------------- */
async function loadJSON(key, shared, fallback) {
  try {
    const r = await window.storage.get(key, shared);
    return r ? JSON.parse(r.value) : fallback;
  } catch {
    return fallback;
  }
}
async function saveJSON(key, value, shared) {
  try {
    await window.storage.set(key, JSON.stringify(value), shared);
  } catch {
    /* best effort */
  }
}

/* ---------------- small UI atoms ---------------- */

function Pill({ active, color, children, onClick }) {
  return (
    <button
      onClick={onClick}
      className="pill"
      style={{
        borderColor: active ? color : "rgba(184,179,217,0.35)",
        background: active ? color + "22" : "transparent",
        color: active ? "#F5F3FF" : "#B8B3D9",
      }}
    >
      {children}
    </button>
  );
}

function CatTag({ id, small }) {
  const c = catById(id);
  const Icon = c.icon;
  return (
    <span className="cat-tag" style={{ color: c.color, borderColor: c.color + "55", fontSize: small ? 11 : 12 }}>
      <Icon size={small ? 11 : 13} strokeWidth={2.2} />
      {c.label}
    </span>
  );
}

function Toast({ message, onDone }) {
  useEffect(() => {
    const t = setTimeout(onDone, 2400);
    return () => clearTimeout(t);
  }, [onDone]);
  return (
    <div className="toast">
      <CheckCircle2 size={16} color="#4ECCA3" />
      {message}
    </div>
  );
}

function Orbit({ onPick }) {

/* ---------------- Mentorship ---------------- */

  const [form, setForm] = useState({ name: "", role: "", cat: "music", bio: "" });

    e.preventDefault();
    if (!form.name.trim() || !form.bio.trim()) return;

    setForm({ name: "", role: "", cat: "music", bio: "" });
    setTab("find");
  };

  return (
    <div className="section">
      <div className="section-head">
        <h2>Mentorship Hub</h2>
        <p className="muted">Guidance, one conversation at a time.</p>
      </div>

        <button className={"tab" + (tab === "become" ? " active" : "")} onClick={() => setTab("become")}>Become a mentor</button>
      </div>

      {tab === "find" ? (
        <div className="creator-grid">
          {mentors.map((m) => (
            <div key={m.id} className="creator-card">
              <div className="creator-card-top">
                <div className="avatar" style={{ background: catById(m.cat).color + "22", color: catById(m.cat).color }}>{m.name[0]}</div>

/* ---------------- Community ---------------- */

function Community({ posts, addPost, myProfile }) {
  const [text, setText] = useState("");
  const [cat, setCat] = useState("music");

  const submit = (e) => {
    e.preventDefault();
    if (!text.trim()) return;
    addPost({ id: uid(), author: myProfile?.name || "You", cat, text: text.trim(), ts: Date.now() });
    setText("");
  };

  const timeAgo = (ts) => {
    const mins = Math.round((Date.now() - ts) / 60000);
    if (mins < 60) return `${mins}m ago`;
    const hrs = Math.round(mins / 60);
    if (hrs < 24) return `${hrs}h ago`;
    return `${Math.round(hrs / 24)}d ago`;
  };

  return (
    <div className="section">
      <div className="section-head">
        <h2>Community</h2>
        <p className="muted">Share work in progress. Find your next collaborator.</p>
      </div>

      <form className="post-form" onSubmit={submit}>
        <textarea value={text} onChange={(e) => setText(e.target.value)} placeholder="Share a post, ask for feedback, or find a collaborator…" />
        <div className="post-form-row">
          <select value={cat} onChange={(e) => setCat(e.target.value)}>
            {CATS.map((c) => <option key={c.id} value={c.id}>{c.label}</option>)}
          </select>
          <button className="btn btn-gold btn-sm" type="submit"><Send size={13} /> Post</button>
        </div>
      </form>

      <div className="post-list">
        {posts.slice().sort((a, b) => b.ts - a.ts).map((p) => (
          <div key={p.id} className="post-card">
            <div className="creator-card-top">
              <div className="avatar" style={{ background: catById(p.cat).color + "22", color: catById(p.cat).color }}>{p.author[0]}</div>
              <div>
                <div className="creator-name">{p.author}</div>
                <div className="creator-loc muted">{timeAgo(p.ts)}</div>
              </div>
            </div>
            <CatTag id={p.cat} small />
            <p className="creator-bio">{p.text}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

/* ---------------- Profile modal ---------------- */

function ProfileModal({ initial, onClose, onSave }) {
  const [form, setForm] = useState(initial || { name: "", location: "", cat: "music", bio: "", portfolio: "", achievements: "" });
  const submit = (e) => {
    e.preventDefault();
    if (!form.name.trim() || !form.bio.trim()) return;
    onSave({ id: initial?.id || uid(), ...form });
  };
  return (
    <div className="modal-backdrop" onClick={onClose}>
      <div className="modal" onClick={(e) => e.stopPropagation()}>
        <div className="modal-head">
          <h3>{initial ? "Edit your profile" : "Join CHU"}</h3>
          <button className="icon-btn" onClick={onClose}><X size={18} /></button>
        </div>
        <form className="form-card" onSubmit={submit}>
          <label>Name
            <input value={form.name} onChange={(e) => setForm({ ...form, name: e.target.value })} placeholder="Your name" required />
          </label>
          <label>Location
            <input value={form.location} onChange={(e) => setForm({ ...form, location: e.target.value })} placeholder="City, Country" />
          </label>
          <label>Talent category
            <select value={form.cat} onChange={(e) => setForm({ ...form, cat: e.target.value })}>
              {CATS.map((c) => <option key={c.id} value={c.id}>{c.label}</option>)}
            </select>
          </label>
          <label>Biography
            <textarea value={form.bio} onChange={(e) => setForm({ ...form, bio: e.target.value })} placeholder="Tell the universe who you are" required />
          </label>
          <label>Portfolio link
            <input value={form.portfolio} onChange={(e) => setForm({ ...form, portfolio: e.target.value })} placeholder="yourwork.com" />
          </label>
          <label>Achievements
            <input value={form.achievements} onChange={(e) => setForm({ ...form, achievements: e.target.value })} placeholder="Awards, features, milestones" />
          </label>
          <button className="btn btn-gold" type="submit"><Sparkles size={15} /> {initial ? "Save changes" : "Create my profile"}</button>
        </form>
      </div>
    </div>
  );
}

        .toast { position:fixed; bottom:24px; left:50%; transform:translateX(-50%); background: var(--surface2); border:1px solid rgba(78,204,163,0.4); color: var(--paper); padding:11px 18px; border-radius:12px; display:flex; align-items:center; gap:8px; font-size:13.5px; z-index:200; box-shadow: 0 8px 30px rgba(0,0,0,0.4); }

        footer.chu-footer { max-width:1180px; margin:0 auto; padding: 30px 20px 50px; display:flex; justify-content:space-between; align-items:center; color: var(--mist); font-size:12.5px; border-top:1px solid rgba(184,179,217,0.1); margin-top:20px; flex-wrap:wrap; gap:10px; }
      `}</style>

      <NavBar view={view} setView={setView} onJoin={() => setShowProfileModal(true)} mobileOpen={mobileOpen} setMobileOpen={setMobileOpen} />

      {view === "home" && <Home setView={setView} setDiscoverFilter={setDiscoverFilter} onJoin={() => (myProfile ? setView("dashboard") : setShowProfileModal(true))} myProfile={myProfile} />}
      {view === "discover" && <Discover creators={creators} filter={discoverFilter} setFilter={setDiscoverFilter} onConnect={handleConnect} connected={connected} />}
      {view === "mentorship" && <Mentorship mentors={mentors} addMentor={handleAddMentor} requested={requested} onRequest={handleRequestMentor} />}
      {view === "opportunities" && <Opportunities opportunities={opportunities} saved={saved} onSave={handleSaveOpp} onApply={handleApplyOpp} applied={applied} />}
      {view === "community" && <Community posts={posts} addPost={handleAddPost} myProfile={myProfile} />}
      {view === "academy" && <Academy courses={courses} />}
      {view === "dashboard" && (
        <Dashboard
          myProfile={myProfile}
          onEditProfile={() => setShowProfileModal(true)}
          savedOpportunities={saved}
          opportunities={opportunities}
          connected={connected}
          requested={requested}
          setView={setView}
        />
      )}

      <footer className="chu-footer">
        <span>Creative Humanity Universe · Where creativity meets opportunity.</span>
        <span>Founded by Felix Samahanda, Zambia</span>
      </footer>

      {showProfileModal && (
        <ProfileModal initial={myProfile} onClose={() => setShowProfileModal(false)} onSave={handleSaveProfile} />
      )}
      {toast && <Toast message={toast} onDone={() => setToast(null)} />}
    </div>
  );
}
